// ===== ADMIN AUTH =====
const ADMIN_USER = "admin";
const ADMIN_PASS = "1234";

function login() {
  const user = document.getElementById("username").value;
  const pass = document.getElementById("password").value;

  if (user === ADMIN_USER && pass === ADMIN_PASS) {
    localStorage.setItem("nextgen_admin", "true");
    showDashboard();
    showToast("Welcome, Admin! 🔐", "success");
  } else {
    showToast("Invalid credentials!", "error");
  }
}

function logout() {
  localStorage.removeItem("nextgen_admin");
  location.reload();
}

function checkAuth() {
  if (localStorage.getItem("nextgen_admin") === "true") {
    showDashboard();
  }
}

function showDashboard() {
  document.getElementById("loginSection").style.display = "none";
  document.getElementById("adminDashboard").style.display = "block";
  document.getElementById("logoutBtn").style.display = "inline";
  loadAdminData();
}

// ===== SHARED DATA =====
function getProducts() {
  return JSON.parse(localStorage.getItem("nextgen_products")) || [];
}

function saveProducts(data) {
  localStorage.setItem("nextgen_products", JSON.stringify(data));
}

// ===== ADMIN STATS =====
function updateStats() {
  const products = getProducts();
  const totalProducts = products.length;
  const totalStock = products.reduce((sum, p) => sum + p.stock, 0);
  const totalValue = products.reduce((sum, p) => sum + (p.price * p.stock), 0);

  document.getElementById("totalProducts").textContent = totalProducts;
  document.getElementById("totalStock").textContent = totalStock;
  document.getElementById("totalValue").textContent = `$${totalValue.toLocaleString()}`;
}

// ===== ADD PRODUCT =====
function addProduct() {
  const name = document.getElementById("newName").value.trim();
  const price = parseFloat(document.getElementById("newPrice").value);
  const stock = parseInt(document.getElementById("newStock").value);
  const category = document.getElementById("newCategory").value;
  const image = document.getElementById("newImage").value.trim() || (category === "laptop" ? "💻" : "📱");

  if (!name || isNaN(price) || isNaN(stock)) {
    showToast("Please fill all required fields!", "error");
    return;
  }

  const products = getProducts();
  const newId = products.length > 0 ? Math.max(...products.map(p => p.id)) + 1 : 1;

  products.push({
    id: newId,
    name,
    price,
    stock,
    category,
    image
  });

  saveProducts(products);
  
  // Clear form
  document.getElementById("newName").value = "";
  document.getElementById("newPrice").value = "";
  document.getElementById("newStock").value = "";
  document.getElementById("newImage").value = "";

  loadAdminData();
  showToast("Product added successfully!", "success");
}

// ===== LOAD ADMIN TABLE =====
function loadAdminData() {
  const products = getProducts();
  const tbody = document.getElementById("adminProductList");
  
  tbody.innerHTML = products.map(p => `
    <tr>
      <td>#${p.id}</td>
      <td><div style="font-size: 2rem; text-align: center;">${p.image}</div></td>
      <td>${p.name}</td>
      <td><span class="card-category cat-${p.category}">${p.category}</span></td>
      <td>$<input type="number" id="price-${p.id}" value="${p.price}" min="0"></td>
      <td><input type="number" id="stock-${p.id}" value="${p.stock}" min="0"></td>
      <td>
        <div class="action-btns">
          <button class="save-btn" onclick="updateProduct(${p.id})">💾 Save</button>
          <button class="delete-btn" onclick="deleteProduct(${p.id})">🗑️ Delete</button>
        </div>
      </td>
    </tr>
  `).join('');

  updateStats();
}

// ===== UPDATE PRODUCT =====
function updateProduct(id) {
  const products = getProducts();
  const product = products.find(p => p.id === id);
  
  if (!product) return;

  const newPrice = parseFloat(document.getElementById(`price-${id}`).value);
  const newStock = parseInt(document.getElementById(`stock-${id}`).value);

  if (isNaN(newPrice) || isNaN(newStock)) {
    showToast("Invalid values!", "error");
    return;
  }

  product.price = newPrice;
  product.stock = newStock;

  saveProducts(products);
  updateStats();
  showToast("Product updated!", "success");
}

// ===== DELETE PRODUCT =====
function deleteProduct(id) {
  if (!confirm("Are you sure you want to delete this product?")) return;

  let products = getProducts();
  products = products.filter(p => p.id !== id);
  saveProducts(products);
  loadAdminData();
  showToast("Product deleted!", "success");
}

// ===== TOAST (shared function) =====
function showToast(message, type = "success") {
  const existing = document.querySelector('.toast');
  if (existing) existing.remove();

  const toast = document.createElement('div');
  toast.className = `toast ${type}`;
  toast.innerHTML = type === "success" ? `✅ ${message}` : `❌ ${message}`;
  document.body.appendChild(toast);

  setTimeout(() => {
    toast.style.opacity = '0';
    toast.style.transform = 'translateY(100px)';
    setTimeout(() => toast.remove(), 300);
  }, 3000);
}
